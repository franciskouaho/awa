/*
RÈGLES FIRESTORE TEMPORAIRES POUR LE DÉVELOPPEMENT
Copiez ces règles dans votre console Firebase :
1. Allez dans Firestore Database > Rules
2. Remplacez le contenu par les règles ci-dessous
3. Cliquez sur "Publier"

ATTENTION : Ces règles sont permissives et ne doivent être utilisées qu'en développement !
*/

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Règles temporaires permissives pour le développement
    match /{document=**} {
      allow read, write: if true;
    }
  }
}

/*
RÈGLES DE PRODUCTION (à utiliser plus tard) :

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Collection users - accès restreint aux propriétaires
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null;
    }
    
    // Autres collections selon vos besoins
    match /prayers/{prayerId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Par défaut, tout refuser
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
